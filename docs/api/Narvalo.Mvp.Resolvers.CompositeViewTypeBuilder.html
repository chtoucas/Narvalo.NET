Class CompositeViewTypeBuilder

To support composite views, we dynamically emit a type which takes multiple views, and exposes them as a single view of the same interface. It's something like this:

public class TestViewComposite : CompositeView<ITestView>, ITestView
{
    public TestViewModel Model
    {
        get { return Views.First().Model; }
        set
        {
            foreach (var view in Views) {
                view.Model = value;
            }
        }
    }

    public event EventHandler Searching
    {
        add
        {
            foreach (var view in Views) {
                view.Searching += value;
            }
        }
        remove
        {
            foreach (var view in Views) {
                view.Searching -= value;
            }
        }
    }
}
